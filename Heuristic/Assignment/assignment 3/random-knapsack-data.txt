import random

cap = random.randrange(70, 100)

items = []

for i in range(30):
    itemno = f"Item {i + 1}"
    itemweight = random.randrange(10, 60)
    itemval = random.randrange(1,100)
    item = tuple([itemno, itemweight, itemval])
    items.append(item)
    
items = tuple(items)

print(f"cap = {cap}")
print(f"items = {items}")
